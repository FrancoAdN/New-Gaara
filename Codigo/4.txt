CREATE TABLE libro (id_libro INT NOT NUL AUTO-INCREMENT,
             PRIMARY KEY(id_libro),
             Titulo VARCHAR(25),
             Editorial VARCHAR(25),
             Area VARCHAR(25),
);

CREATE TABLE libaut (
               FOREING KEY(id_libro) REFERENCES Libro(id_libro),
               FOREING KEY(id_autor) REFERENCES Autor(id_autor)
);

CREATE TABLE autor(id_autor INT NOT NULL AUTO-INCREMENT,
             PRIMARY KEY(id_autor),
             Nombre VARCHAR(25),
             Nacionalidad VARCHAR(25)
);

CREATE TABLE prestamo(fecha_prestamo DATE NOT NULL AUTO-INCREMENT,
             PRIMARY KEY(fecha_prestamo),
             FOREING KEY(id_libro) REFERENCES libro(id_libro),
             FOREING KEY(id_lector) REFERENCES prestamo(id_lector),
             Fecha_devolucion DATE,
             Devuelto INT
);

CREATE TABLE estudiante(id_lector INT NOT NULL AUTO-INCREMENT,
             PRIMARY KEY(id_lector),
             CI INT,
             Nombre VARCHAR(25),
             Direccion NVARCHAR(25),
             Carrera VARCHAR(25),
             Edad INT}
);
