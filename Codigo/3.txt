CREATE TABLE almacenes (codigo INT NOT NULL AUTO-INCREMENT,
             PRIMARY KEY(codigo),
             Lugar NVARCHAR(100),
             Capacidad INT
);

CREATE TABLE cajas (num_referencia CHAR(5),
             PRIMARY KEY(num_referencia),
             Contenido NVARCHAR(100),
             Valor INT,
             Codigo_almacen INT
             FOREIGN KEY Codigo_almacen (Codigo_almacen) REFERENCES cajas(Codigo_almacen)
);
